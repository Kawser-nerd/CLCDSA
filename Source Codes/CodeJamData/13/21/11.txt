#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<math.h>

int cmp(const void* a, const void *b){
	 return (*(int*)a-*(int*)b);
}

int main(){
	int k;
	scanf("%d",&k);
	for(int t=0;t<k;t++){
		int size,n;
		int a[105];
		int b[105];
		int c[105];
		scanf("%d%d",&size,&n);
		for(int i=0;i<n;i++)scanf("%d",&a[i]);
		qsort(a,n,sizeof(int),cmp);
		//for(int i=0;i<n;i++)printf("%d ",a[i]);printf("\n");
		int min=n;
		c[0]=0;
		b[0]=size;
		if(size==1)
		{
				printf("Case #%d: %d\n",t+1,n);
				continue;
		}
		for(int i=1;i<=n;i++){
			c[i]=c[i-1];
			b[i]=i>0?b[i-1]:size;
			while((b[i]<=a[i-1])){
				b[i]+=b[i]-1;
				c[i]++;
			}
			b[i]+=a[i-1];
			if(c[i]+n-i<min)min=c[i]+n-i;
		}
		printf("Case #%d: %d\n",t+1,min);
	}
	//while(1);
	return 0;
}
