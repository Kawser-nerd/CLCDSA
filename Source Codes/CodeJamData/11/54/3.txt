#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <algorithm>
#include <queue>
#include <vector>
using namespace std;

#define MAXN 30
typedef long long llint;

llint ans;

void Search(char *s,int len,int lvl,llint val)
{
	if(lvl==len)
	{
		llint tmp=floor(sqrt((double)val));
		if(tmp*tmp==val||(tmp+1)*(tmp+1)==val)
		{
			//printf("%lld\n",val);
			ans=val;
		}
		return;
	}

	if(s[lvl]=='?')
	{
		Search(s,len,lvl+1,val*2);
		Search(s,len,lvl+1,val*2+1);
	}
	else if(s[lvl]=='1')
		Search(s,len,lvl+1,val*2+1);
	else
		Search(s,len,lvl+1,val*2);

}

char buf[100];

void D2B(char *str,llint val)
{
	int i=0;
	while(val)
	{
		if(val&1)
		{
			str[i]='1';
		}
		else
		{
			str[i]='0';
		}
		i++;
		val>>=1;
	}
	str[i]=0;
}

int main()
{
	//freopen("in.txt","r",stdin);
	int t,ct=1;
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%s",buf);
		int len=strlen(buf);
		Search(buf,len,0,0);
		D2B(buf,ans);
		printf("Case #%d: ",ct++);
		len=strlen(buf);
		for(int i=len-1;i>=0;i--)
			printf("%c",buf[i]);
		printf("\n");
	}
	return 0;
}
