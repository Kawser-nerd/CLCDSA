#include <stdio.h>
char c[100][100];
void f(int x,int y)
{
	if(c[x][y]=='#'&&c[x+1][y]=='#'&&c[x][y+1]=='#'&&c[x+1][y+1]=='#')
	{
		c[x][y]='/';
		c[x+1][y+1]='/';
		c[x][y+1]='\\';
		c[x+1][y]='\\';
	}
}
int main()
{
	freopen("A-small-attempt1.in","r",stdin);
	freopen("out.txt","w",stdout);
	int T;
	scanf("%d",&T);
	for(int _=1;_<=T;_++)
	{
		int r,C;
		scanf("%d%d",&r,&C);
		int i,j;
		for(i=0;i<r;i++)
			scanf("%s",c[i]);
		printf("Case #%d:\n",_);
		int flag=0;
		for(i=0;i<r-1;i++)
		{
			for(j=0;j<C-1;j++)
			{
				f(i,j);
			}
		}
		for(i=0;i<r;i++)
			for(j=0;j<C;j++)
				if(c[i][j]=='#')
					flag=1;
		if(flag)
			printf("Impossible\n");
		else
		{
			for(i=0;i<r;i++)
				printf("%s\n",c[i]);
		}
	}
	return 0;
}
