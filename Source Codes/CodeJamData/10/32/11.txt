#include <stdio.h>
int pow(int a,int b)
{
	int i,j,k;
	int res=1;
	for(i=0;i<b;i++)
	{
		res*=a;
	}
	return res;
}
int cal1(int l,int c,int p)
{
	int i,j,k;
	int res=l;
	for(i=0;;i++)
	{
		if(res>=p) return i;
		res*=c;
	}
}
