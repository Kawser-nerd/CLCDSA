#include <stdio.h>
#include <string.h>
long long gcd(long long x,long long y)
{
	if (y==0) return x;
	else return gcd(y,x%y);
}
int main()
{
	freopen("d:\\1.in","r",stdin);
	freopen("d:\\1.txt","w",stdout);
	int t;
	scanf("%d",&t);
    long long  x,y;
	for (int k=1;k<=t;k++)
	{
		scanf("%lld/%lld",&x,&y);
		long long tmp = gcd(x,y);
		x = x / tmp;
		y = y / tmp;
		long long p = 1;
		x = x >> 1;
		while (x)
		{
			x = x>>1;
			p = p<<1;
		}
		tmp = gcd(p,y);
		y = y / tmp;
		long long ans = 0;
		while (y!=1)
		{
			if (y&1) {ans = -1;break;}
			y = y >> 1;
			ans ++;
		}
		if (ans==-1) printf("Case #%d: impossible\n",k);
		else printf("Case #%d: %d\n",k,ans);
	}
	return 0;
}
