int okay(int x,int y,int r)
{
	int i,j,k;
	if(x<0) return 0;
	if(x+r>_m) return 0;
	if(y<0) return 0;
	if(y+r>_n) return 0;
	for(i=x;i<x+r;i++)
	{
		for(j=y;j<y+r;j++)
		{
			if(used[i][j]) return 0;
		}
	}
	int pre=bm[x][y];
	for(i=x+1;i<x+r;i++)
	{
		if(bm[i][y]==pre) return 0;
		pre=(pre==0?1:0);
	}
	for(i=x;i<x+r;i++)
	{
		int pre=bm[i][y];
		for(j=y+1;j<y+r;j++)
		{
			if(bm[i][j]==pre) return 0;
			pre=(pre==0?1:0);
		}
	}
	return 1;
}

}
